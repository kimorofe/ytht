    <script>
        // إنشاء مفتاح فريد لكل صفحة بناءً على عنوان URL
        const pageKey = `familyRecords_${window.location.href}`;

        // تحميل السجلات المخزنة من Local Storage
        let records = JSON.parse(localStorage.getItem(pageKey)) || [];

        // تحديث عدد السجلات المحفوظة
        function updateRecordsCount() {
            const recordsCount = document.getElementById('recordsCount');
            recordsCount.textContent = `عدد السجلات المحفوظة: ${records.length}`;
        }

        // حفظ البيانات في Local Storage
        function saveData() {
            const form = document.getElementById('familyForm');
            const formData = new FormData(form);
            const record = {};
            formData.forEach((value, key) => {
                record[key] = value;
            });
            records.push(record);
            localStorage.setItem(pageKey, JSON.stringify(records));
            showMessage("تم حفظ البيانات بنجاح!", "success");
            form.reset();
            updateRecordsCount();
        }

        // عرض السجلات
        function viewRecords() {
            const recordsDisplay = document.getElementById('recordsDisplay');
            recordsDisplay.innerHTML = '';

            if (records.length === 0) {
                showMessage("لا توجد سجلات للعرض.", "error");
                return;
            }

            const summary = document.createElement('div');
            summary.className = 'records-summary';

            records.forEach((record, index) => {
                const item = document.createElement('div');
                item.className = 'record-item';

                const info = document.createElement('div');
                info.textContent = `رمز الأسرة: ${record.familyCode}, الزوج: ${record.husbandName}, الزوجة: ${record.wifeName}`;

                const viewButton = document.createElement('button');
                viewButton.textContent = 'عرض التفاصيل';
                viewButton.onclick = () => viewDetailedRecord(index);

                const deleteButton = document.createElement('button');
                deleteButton.textContent = 'حذف';
                deleteButton.className = 'delete-button';
                deleteButton.onclick = () => deleteRecord(index);

                item.appendChild(info);
                item.appendChild(viewButton);
                item.appendChild(deleteButton);
                summary.appendChild(item);
            });

            // زر إغلاق عرض السجلات
            const closeButton = document.createElement('button');
            closeButton.textContent = 'إغلاق عرض السجلات';
            closeButton.className = 'close-records-button';
            closeButton.onclick = () => {
                recordsDisplay.innerHTML = ''; // إخفاء السجلات
            };

            recordsDisplay.appendChild(summary);
            recordsDisplay.appendChild(closeButton);
        }

        // حذف سجل
        function deleteRecord(index) {
            records.splice(index, 1);
            localStorage.setItem(pageKey, JSON.stringify(records));
            showMessage("تم حذف السجل بنجاح!", "success");
            updateRecordsCount();
            viewRecords(); // تحديث عرض السجلات بعد الحذف
        }

        // عرض تفاصيل السجل
        function viewDetailedRecord(index) {
            const record = records[index];
            const recordsDisplay = document.getElementById('recordsDisplay');
            recordsDisplay.innerHTML = '';

            const detailed = document.createElement('div');
            detailed.className = 'detailed-record';

            for (const key in record) {
                const paragraph = document.createElement('p');
                paragraph.textContent = `${key}: ${record[key]}`;
                detailed.appendChild(paragraph);
            }

            const closeButton = document.createElement('button');
            closeButton.textContent = 'إغلاق';
            closeButton.className = 'close-button';
            closeButton.onclick = () => viewRecords();

            detailed.appendChild(closeButton);
            recordsDisplay.appendChild(detailed);
        }

        // البحث في السجلات
        function searchRecord() {
            const searchTerm = document.getElementById('search').value.toLowerCase();
            const filteredRecords = records.filter(record => 
                record.familyCode.toLowerCase().includes(searchTerm) ||
                record.husbandName.toLowerCase().includes(searchTerm) ||
                record.wifeName.toLowerCase().includes(searchTerm)
            );

            const recordsDisplay = document.getElementById('recordsDisplay');
            recordsDisplay.innerHTML = '';

            if (filteredRecords.length === 0) {
                showMessage("لم يتم العثور على سجلات.", "error");
                return;
            }

            const summary = document.createElement('div');
            summary.className = 'records-summary';

            filteredRecords.forEach((record, index) => {
                const item = document.createElement('div');
                item.className = 'record-item';

                const info = document.createElement('div');
                info.textContent = `رمز الأسرة: ${record.familyCode}, الزوج: ${record.husbandName}, الزوجة: ${record.wifeName}`;

                const viewButton = document.createElement('button');
                viewButton.textContent = 'عرض التفاصيل';
                viewButton.onclick = () => viewDetailedRecord(index);

                const deleteButton = document.createElement('button');
                deleteButton.textContent = 'حذف';
                deleteButton.className = 'delete-button';
                deleteButton.onclick = () => deleteRecord(index);

                item.appendChild(info);
                item.appendChild(viewButton);
                item.appendChild(deleteButton);
                summary.appendChild(item);
            });

            // زر إغلاق عرض السجلات
            const closeButton = document.createElement('button');
            closeButton.textContent = 'إغلاق عرض السجلات';
            closeButton.className = 'close-records-button';
            closeButton.onclick = () => {
                recordsDisplay.innerHTML = ''; // إخفاء السجلات
            };

            recordsDisplay.appendChild(summary);
            recordsDisplay.appendChild(closeButton);
        }

        // عرض الرسائل
        function showMessage(message, type) {
            const messageContainer = document.getElementById('messageContainer');
            messageContainer.innerHTML = '';

            const messageDiv = document.createElement('div');
            messageDiv.className = type === 'success' ? 'message success' : 'message error';
            messageDiv.textContent = message;
            messageContainer.appendChild(messageDiv);

            setTimeout(() => {
                messageDiv.remove();
            }, 3000);
        }

        // تحديث عدد السجلات عند تحميل الصفحة
        window.onload = function () {
            updateRecordsCount(); // تحديث عدد السجلات فقط دون عرضها
        };
    </script>
